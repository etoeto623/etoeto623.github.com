git提交时自动生成模版msg

#!/usr/bin/env groovy
 
import static java.lang.System.exit
import static java.lang.System.getenv
import groovy.json.JsonSlurper
import groovy.json.JsonBuilder
import java.util.LinkedHashMap
 
if(null == args || args.length == 0){
    println "参数错误"
    exit(1)
}
 
if('--help'.equals(args[0])){
    println """Usage:
    gcommit commit_msg [taskId]
应用分支对应的taskId在用户home目录下的gcommit.json文件中配置"""
    exit(0)
}
 
// 获取home目录
def getHomeDir(){
    getenv "HOME"
}
 
Map<String, Map<String, String>> getConfigMap(){
    def cfgStr = new File(homeDir+"/gcommit.json").text
    JsonSlurper parser = new JsonSlurper();
    Map<String, Map<String, Integer>> config = parser.parseText(cfgStr)
    return config
}
 
def taskId(String app, branch){
    Map<String, Map<String, Integer>> config = configMap
    if(!config.containsKey(app) || !config[app].containsKey(branch)){
        println "请先配置git分支对应的taskId"
        exit(1)
    }
    config[app][branch]
}
 
// 读取当前分支名称
def getGitBranch(){
    new File(".git/HEAD").text.replace('ref: refs/heads/', '').trim()
}
 
// 获取应用名
def getAppName(){
    'pwd'.execute().text.split('/').last().trim()
}
 
if(args[0].equals("-setTask")){
    // 设置taskId
    if(args.length < 2){
        println "请指定taskId"
        exit(1)
    }
    def taskId = Integer.parseInt(args[1])
    def apName = appName
    def branch = gitBranch
    Map<String, Map<String, Integer>> cfMap = configMap
    if(!cfMap.containsKey(apName)){
        cfMap.put(apName, new LinkedHashMap<String, Integer>());
    }
    cfMap.get(apName).put(branch, taskId)
    new File(homeDir+"/gcommit.json").text = new JsonBuilder(cfMap).toPrettyString()
}else{
    def task = args.length > 1 ? args[1] : taskId(appName,gitBranch)
    println "git commit -m Desc:${args[0]} -m Type:ReqDev -m ID:$task ".execute().text
}

推送到gerrit

#!/bin/bash
branch_name="$1"
if [ "" == "$1" ]; then
  # 没有传分支号，则获取当前分支名
  branch_name=`git branch | grep \* | cut -d ' ' -f2`
fi
function echo_green(){
    echo -e "\033[36m$1 \033[0m"
}
echo_green =============================================
echo Starting push $branch_name to gerrit
git push origin $branch_name:refs/for/$branch_name
echo Push finished
echo_green =============================================

jvm占用内存统计

nohup cmd >/dev/null 2>&1 &

#!/bin/bash
 
function log_memo(){
  pid=`jps -l | grep tj.jar | awk '{print $1}'`
  real_use=`ps -e -o 'pid,comm,rsz' | grep java | grep $pid | awk '{print strftime("[%Y-%m-%d %H:%M:%S]",systime()),$NF,$NF/1024/1024 "G"}'`
  jstat_use=''
  if(($1 % 2 == 0));
  then
    jstat_use=`jstat -gc $pid | awk '{if(NR==2){print $3,$4,$6,$8,$10,$12}}'`
  fi
  echo $real_use $jstat_use >> memo_log_`date +%m%d`.txt
}
 
loop_times=0
while ((loop_times < 1200))
do
  log_memo $loop_times
  ((loop_times += 1))
  sleep 3m
done

记录pmap信息

#!/bin/bash
 
function logPmap(){
  pid=`jps -l | grep tj.jar | awk '{print $1}'`
  dateStr=`date +%m%d%H%M`
  pmap -x $pid | awk '{if(NR>2&&$3>60000&&$3<100000)print $0}' > /home/admin/tj/pmaplogs/`echo $dateStr`.txt
  ps -e -o 'pid,comm,rsz' | grep java | grep $pid | awk '{print strftime("[%Y-%m-%d %H:%M:%S]",systime()),$NF,$NF/1024/1024 "G"}' >> /home/admin/tj/pmaplogs/`echo $dateStr`.txt
}
 
loop_times=0
while ((loop_times < 200))
do
  logPmap
  ((loop_times += 1))
  sleep 1h
done
