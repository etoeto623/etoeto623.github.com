go语言实现抢红包

  2021-07-01   红包 字数统计: 148字 阅读时长: 1分 
package main

import (
	"fmt"
	"math/rand"
	"os"
	"sort"
	"strconv"
	"time"
)

func main(){
	args := os.Args
	if len(args) != 3 {
		fmt.Println("wrong argument")
		return
	}

	amount, err := strconv.ParseInt(args[1], 10, 32)
	if nil != err {
		fmt.Println("wrong argument: amount")
		return
	}
	total, err := strconv.ParseInt(args[2], 10, 32)
	if nil != err {
		fmt.Println("wrong argument: total")
	}

	fmt.Printf("amount:%d, total:%d\n", amount, total)

	maxRand := int(amount * 100)

	points := []int{}

	r := rand.New(rand.NewSource(time.Now().Unix()))

	for {
		if int64(len(points)) >= total-1{
			break
		}
		randN := r.Intn(maxRand-1) + 1
		if !contains(points, randN){
			points = append(points, randN)
		}
	}

	sort.Ints(points)

	pre := 0
	packs := []float32{}
	for idx:=0; idx < len(points); idx++ {
		packs = append(packs, float32(points[idx] - pre)/100)
		pre = points[idx]
	}
	packs = append(packs, float32(maxRand-pre)/100)

	fmt.Println(packs)
}

func contains(arr []int, num int) bool {
	for _,v := range arr {
		if v == num {
			return true
		}
	}
	return false
}
